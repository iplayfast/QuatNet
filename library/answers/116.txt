from collections import defaultdict

def compute_topological_orderings(graph):
    def dfs(node):
        visited[node] = 'visiting'
        for neighbor in graph[node]:
            if visited[neighbor] == 'unvisited':
                dfs(neighbor)
        visited[node] = 'visited'
        order.appendleft(node)

    def backtrack(path=[]):
        if len(path) == len(graph):
            result.append(path[:])
            return
        node = path[-1]
        for neighbor in graph[node]:
            if visited[neighbor] == 'visited':
                backtrack(path + [neighbor])

    nodes = set()
    for u, vs in graph.items():
        nodes.update(vs)
    visited = defaultdict(lambda: 'unvisited')
    order = collections.deque()

    for node in nodes:
        if visited[node] == 'unvisited':
            dfs(node)

    result = []
    for node in order:
        visited[node] = 'visited'
        backtrack([node])

    return result

# Example usage:
graph = {
    0: [1, 2],
    1: [3],
    2: [3],
    3: []
}
print(compute_topological_orderings(graph))
