```python
def solve():
    def max_sum_subarray(nums):
        if not nums:
            return 0
        
        current_max = nums[0]
        global_max = nums[0]
        
        for num in nums[1:]:
            current_max = max(num, current_max + num)
            global_max = max(global_max, current_max)
        
        return global_max

    # Example usage
    input_nums = [1, -2, 3, 5, -3, 2]
    result = max_sum_subarray(input_nums)
    print(result)
```
