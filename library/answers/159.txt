class DynamicArray:
    def __init__(self, initial_capacity=1):
        self._capacity = initial_capacity
        self._size = 0
        self._array = [None] * self._capacity
    
    def __len__(self):
        return self._size
    
    def __getitem__(self, index):
        if index < 0:
            index += self._size
        if not 0 <= index < self._size:
            raise IndexError("Index out of range")
        return self._array[index]
    
    def __setitem__(self, index, value):
        if index < 0:
            index += self._size
        if not 0 <= index < self._size:
            raise IndexError("Index out of range")
        self._array[index] = value
    
    def append(self, value):
        if self._size == self._capacity:
            self._resize(2 * self._capacity)
        self._array[self._size] = value
        self._size += 1
    
    def _resize(self, new_capacity):
        new_array = [None] * new_capacity
        for i in range(self._size):
            new_array[i] = self._array[i]
        self._array = new_array
        self._capacity = new_capacity
    
    def extend(self, iterable):
        for item in iterable:
            self.append(item)
    
    def pop(self, index=-1):
        if self._size == 0:
            raise IndexError("Pop from empty array")
        
        if index < 0:
            index += self._size
        if not 0 <= index < self._size:
            raise IndexError("Index out of range")
        
        value = self._array[index]
        
        # Shift elements left if not the last element
        if index != self._size - 1:
            for i in range(index, self._size - 1):
                self._array[i] = self._array[i + 1]
        
        self._array[self._size - 1] = None
        self._size -= 1
        
        # Optional: shrink array if usage is too low
        if 0 < self._size <= self._capacity // 4 and self._capacity > 1:
            self._resize(max(1, self._capacity // 2))
        
        return value
    
    def __iter__(self):
        for i in range(self._size):
            yield self._array[i]
    
    def __repr__(self):
        return f"DynamicArray([{', '.join(repr(x) for x in self)}])"
