```python
import random

choices = ['rock', 'paper', 'scissors']

def get_human_choice():
    while True:
        choice = input("Enter your choice (rock/paper/scissors): ").lower()
        if choice in choices:
            return choice
        else:
            print("Invalid choice. Please enter rock, paper, or scissors.")

def get_computer_choice(history):
    most_common = max(set(history), key=history.count) if history else random.choice(choices)
    return most_common

def determine_winner(player, computer):
    if player == computer:
        return "It's a tie!"
    elif (player == 'rock' and computer == 'scissors') or \
         (player == 'scissors' and computer == 'paper') or \
         (player == 'paper' and computer == 'rock'):
        return "You win!"
    else:
        return "Computer wins."

def play_game():
    history = []
    while True:
        human_choice = get_human_choice()
        computer_choice = get_computer_choice(history)
        history.append(computer_choice)

        print(f"You chose: {human_choice}")
        print(f"Computer chose: {computer_choice}")

        result = determine_winner(human_choice, computer_choice)
        print(result)

if __name__ == "__main__":
    play_game()
```
