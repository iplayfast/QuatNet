def kmp_search(text, pattern):
    n, m = len(text), len(pattern)
    lps = [0] * m
    j = 0
    for i in range(1, m):
        while j > 0 and pattern[i] != pattern[j]: j = lps[j - 1]
        if pattern[i] == pattern[j]: j += 1; lps[i] = j
    j = 0
    for i in range(n):
        while j > 0 and text[i] != pattern[j]: j = lps[j - 1]
        if text[i] == pattern[j]: j += 1
        if j == m: return i - m + 1
    return -1
