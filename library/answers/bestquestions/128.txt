import hashlib

class CountMinSketch:
    def __init__(self, width, depth):
        self.width = width
        self.depth = depth
        self.counts = [[0] * width for _ in range(depth)]

    def _hashes(self, item):
        return [int(hashlib.md5(f'{item}{i}'.encode()).hexdigest(), 16) % self.width for i in range(self.depth)]

    def add(self, item, count=1):
        for i, h in enumerate(self._hashes(item)):
            self.counts[i][h] += count

    def estimate(self, item):
        return min(self.counts[i][h] for i, h in enumerate(self._hashes(item)))
