import time

def stopwatch():
    input('Press Enter to start...')
    start = time.time()
    lap_num = 1
    while True:
        try:
            input(f'Lap {lap_num}: {time.time() - start:.2f}s - Press Enter for next lap...')
            lap_num += 1
        except KeyboardInterrupt:
            print(f'\nTotal: {time.time() - start:.2f}s')
            break
