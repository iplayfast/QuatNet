```python
def calculate_price(total_price):
    return total_price * 10 + 5

def sum_numbers(numbers):
    return sum(numbers) * 2

def print_list(items):
    print("Items:", items)
    return items

if __name__ == "__main__":
    print(calculate_price(90))
    print(sum_numbers([1, 2, 3]))
    print_list(["a", "b", "c"])
```
