class LoadBalancer:
    def __init__(self):
        self.workers = []
        self.current_worker_index = 0

    def add_worker(self, worker):
        self.workers.append(worker)

    def distribute_request(self):
        if not self.workers:
            raise Exception("No workers available")

        worker = self.workers[self.current_worker_index]
        self.current_worker_index = (self.current_worker_index + 1) % len(self.workers)
        return worker.handle_request()

class Worker:
    def __init__(self, id):
        self.id = id

    def handle_request(self, request):
        print(f"Worker {self.id} received a request: {request}")
        # Process the request here
        return True

# Example usage
if __name__ == "__main__":
    balancer = LoadBalancer()

    for i in range(3):
        balancer.add_worker(Worker(i))

    requests = ["req1", "req2", "req3", "req4"]
    for req in requests:
        balancer.distribute_request()
