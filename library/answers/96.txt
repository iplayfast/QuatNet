```python
import math

def factorial(n: int) -> int:
    """
    Calculates the factorial of a non-negative integer n (n!).

    Args:
        n: The non-negative integer.

    Returns:
        The factorial of n.

    Raises:
        ValueError: If n is negative or not an integer.
    """
    if not isinstance(n, int):
        raise ValueError("Input must be an integer.")
    if n < 0:
        raise ValueError("Factorial is only defined for non-negative integers.")
    
    if n == 0:
        return 1
    
    result = 1
    for i in range(1, n + 1):
        result *= i
    return result

# Example Usage and Testing
if __name__ == "__main__":
    test_cases = [
        (0, 1),
        (1, 1),
        (5, 120),
        (10, 3628800)
    ]
    
    print("--- Testing Valid Inputs ---")
    for n, expected in test_cases:
        actual = factorial(n)
        print(f"Factorial of {n}: {actual} (Expected: {expected})")
        assert actual == expected

    print("\n--- Testing Edge Cases and Errors ---")
    
    # Test negative input
    try:
        factorial(-1)
    except ValueError as e:
        print(f"Caught expected error for -1: {e}")

    # Test float input
    try:
        factorial(5.5)
    except ValueError as e:
        print(f"Caught expected error for 5.5: {e}")
```
