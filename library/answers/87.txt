def sieve_of_eratosthenes():
    """Yield prime numbers using the Sieve of Eratosthenes algorithm."""
    D = {}  # dictionary to hold multiples of primes
    q = 2   # start checking from the first prime number

    while True:
        if q not in D:  # if q is not marked, it is a prime
            yield q
            D[q * q] = [q]  # mark the first multiple of q as its multiples
        else:
            for p in D[q]:  # if q is marked, get its prime factors
                D.setdefault(q + p, []).append(p)
            del D[q]  # remove q from the dictionary

        q += 1
