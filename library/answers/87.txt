def calculate_factorial(n: int) -> int:
    """
    Calculates the factorial of a non-negative integer n (n!).

    Args:
        n: The integer for which to calculate the factorial.

    Returns:
        The factorial of n.

    Raises:
        ValueError: If n is a negative number.
    """
    if not isinstance(n, int):
        raise TypeError("Input must be an integer.")
    if n < 0:
        raise ValueError("Factorial is not defined for negative numbers.")
    
    if n == 0:
        return 1
    
    result = 1
    for i in range(1, n + 1):
        result *= i
    return result

if __name__ == '__main__':
    # Test Case 1: Positive integer
    num1 = 5
    print(f"The factorial of {num1} is: {calculate_factorial(num1)}") # Expected: 120

    # Test Case 2: Zero
    num2 = 0
    print(f"The factorial of {num2} is: {calculate_factorial(num2)}") # Expected: 1

    # Test Case 3: Larger number
    num3 = 10
    print(f"The factorial of {num3} is: {calculate_factorial(num3)}") # Expected: 3628800
    
    # Test Case 4: Error handling (Negative)
    try:
        calculate_factorial(-1)
    except ValueError as e:
        print(f"Caught expected error: {e}")

    # Test Case 5: Error handling (Type)
    try:
        calculate_factorial(5.5)
    except TypeError as e:
        print(f"Caught expected error: {e}")
