def intersect_and_union(set1, set2):
    # Convert lists to dictionaries with values as True
    dict1 = {x: True for x in set1}
    dict2 = {x: True for x in set2}

    # Intersection: elements present in both sets
    intersection = [k for k in dict1 if k in dict2]

    # Union: all unique elements from both sets
    union = list(dict1.keys() | dict2.keys())

    return intersection, union
