from array import array

def ring_buffer(capacity):
    buf = [None] * capacity
    read_pos = 0
    write_pos = 0

    def push(item):
        nonlocal write_pos
        nxt = (write_pos + 1) % capacity
        if nxt == read_pos: return False
        buf[write_pos] = item
        write_pos = nxt
        return True

    def pop():
        nonlocal read_pos
        if read_pos == write_pos: return None
        item = buf[read_pos]
        read_pos = (read_pos + 1) % capacity
        return item

    return push, pop
