def intersect_and_union(set1, set2):
    # Convert lists to dictionaries with unique keys for each element
    dict1 = {item: True for item in set1}
    dict2 = {item: True for item in set2}

    # Compute intersection using dictionary comprehension
    intersection = [item for item in dict1 if item in dict2]

    # Compute union using dictionary keys and list concatenation
    union = list(dict1.keys()) + [item for item in dict2 if item not in dict1]

    return intersection, union
