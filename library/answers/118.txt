```python
import math

def calculate_factorial(n: int) -> int:
    """
    Calculates the factorial of a non-negative integer n (n!).

    Args:
        n: The non-negative integer.

    Returns:
        The factorial of n, or 0 if n is negative.
    """
    if not isinstance(n, int):
        raise TypeError("Input must be an integer.")
        
    if n < 0:
        return 0  # Factorial is not defined for negative numbers
    elif n == 0:
        return 1
    else:
        result = 1
        for i in range(1, n + 1):
            result *= i
        return result

# Example usage:
if __name__ == "__main__":
    num1 = 5
    print(f"The factorial of {num1} is: {calculate_factorial(num1)}")

    num2 = 0
    print(f"The factorial of {num2} is: {calculate_factorial(num2)}")

    num3 = 7
    print(f"The factorial of {num3} is: {calculate_factorial(num3)}")

    num_negative = -3
    print(f"The factorial of {num_negative} is: {calculate_factorial(num_negative)}")

    # Test for type error (optional but good practice)
    try:
        calculate_factorial("a")
    except TypeError as e:
        print(f"Error caught: {e}")
```
