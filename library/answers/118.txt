```python
import collections

def count_char_frequency(text: str) -> dict[str, int]:
    """
    Counts the frequency of each character in the given string.

    Args:
        text: The input string.

    Returns:
        A dictionary where keys are characters and values are their counts.
    """
    if not isinstance(text, str):
        raise TypeError("Input must be a string.")
        
    # Use collections.Counter for an efficient and clean solution
    return dict(collections.Counter(text))

# --- Example Usage ---

if __name__ == "__main__":
    # Test Case 1: Basic sentence
    sample_text_1 = "programming is fun"
    frequency_1 = count_char_frequency(sample_text_1)
    print(f"--- Analysis of '{sample_text_1}' ---")
    print(f"Character Frequencies: {frequency_1}")

    # Test Case 2: Empty string
    sample_text_2 = ""
    frequency_2 = count_char_frequency(sample_text_2)
    print("\n--- Analysis of '' ---")
    print(f"Character Frequencies: {frequency_2}")
    
    # Test Case 3: Punctuation and capitalization sensitivity
    sample_text_3 = "Hello World!"
    frequency_3 = count_char_frequency(sample_text_3)
    print("\n--- Analysis of 'Hello World!' ---")
    print(f"Character Frequencies: {frequency_3}")
```
