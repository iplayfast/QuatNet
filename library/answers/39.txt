def to_pig_latin(text):
    r=[]
    for w in text.split():
        if w[0].isalpha():
            r.append(w[1:]+w[0]+'ay')
        else: r.append(w)
    return ' '.join(r)

def from_pig_latin(text):
    r=[]
    for w in text.split():
        if w.endswith('ay') and len(w)>2:
            r.append(w[-3]+w[:-3])
        else: r.append(w)
    return ' '.join(r)
