def calculate_factorial(n: int) -> int:
    """
    Calculates the factorial of a non-negative integer n (n!).

    Args:
        n: The non-negative integer.

    Returns:
        The factorial of n.

    Raises:
        ValueError: If n is a negative number.
    """
    if not isinstance(n, int):
        raise TypeError("Input must be an integer.")
    if n < 0:
        raise ValueError("Factorial is not defined for negative numbers.")
    if n == 0:
        return 1
    
    result = 1
    for i in range(1, n + 1):
        result *= i
    return result

if __name__ == '__main__':
    # Example 1: Positive integer
    number1 = 5
    print(f"The factorial of {number1} is: {calculate_factorial(number1)}") # Output: 120

    # Example 2: Zero
    number2 = 0
    print(f"The factorial of {number2} is: {calculate_factorial(number2)}") # Output: 1

    # Example 3: Large integer
    number3 = 10
    print(f"The factorial of {number3} is: {calculate_factorial(number3)}") # Output: 3628800

    # Example 4: Testing error handling
    try:
        calculate_factorial(-1)
    except ValueError as e:
        print(f"Error caught (Expected): {e}")
        
    try:
        calculate_factorial(5.5)
    except TypeError as e:
        print(f"Error caught (Expected): {e}")
