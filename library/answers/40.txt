def find_100():
    from itertools import product
    results=[]
    for ops in product(['+','-',''],repeat=8):
        expr='1'
        for i,op in enumerate(ops,2):
            expr+=op+str(i)
        if eval(expr)==100:
            results.append(expr)
    return results
# Finds 11 solutions like '1+2+3-4+5+6+78+9', '12+3-4+5+67+8+9', etc.
