def calculate_factorial(n: int) -> int:
    """
    Calculates the factorial of a non-negative integer n (n!).

    Args:
        n: The non-negative integer.

    Returns:
        The factorial of n.

    Raises:
        ValueError: If n is negative.
        TypeError: If n is not an integer.
    """
    if not isinstance(n, int):
        raise TypeError("Input must be an integer.")
    if n < 0:
        raise ValueError("Factorial is not defined for negative numbers.")
    
    if n == 0:
        return 1
    
    result = 1
    for i in range(1, n + 1):
        result *= i
    
    return result

if __name__ == '__main__':
    # Example usage:
    print(f"Factorial of 5: {calculate_factorial(5)}")
    print(f"Factorial of 0: {calculate_factorial(0)}")
    print(f"Factorial of 7: {calculate_factorial(7)}")

    # Example error handling:
    try:
        calculate_factorial(-1)
    except ValueError as e:
        print(f"\nError caught: {e}")
