```python
import json

def read_and_validate_json(file_path, indent=4):
    try:
        with open(file_path, 'r') as f:
            data = json.load(f)
        print(json.dumps(data, indent=indent))
        return True
    except Exception as e:
        print("Error:", e)
        return False

if __name__ == "__main__":
    import sys
    if len(sys.argv) != 3:
        print("Usage: python script.py <json_file> <indent>")
        exit(1)
    
    file_path = sys.argv[1]
    try:
        indent = int(sys.argv[2])
    except ValueError:
        print("Indent must be an integer.")
        exit(1)

    read_and_validate_json(file_path, indent)
```
