def fibonacci_sequence(n):
    """
    Calculates the Fibonacci sequence up to the n-th term.

    Args:
        n (int): The number of terms to generate (must be non-negative).

    Returns:
        list[int]: A list containing the first n Fibonacci numbers.

    Raises:
        ValueError: If n is negative.
    """
    if n < 0:
        raise ValueError("Input must be a non-negative integer.")
    if n == 0:
        return []
    if n == 1:
        return [0]

    sequence = [0, 1]
    
    # We already have 2 terms, so we iterate n - 2 times
    for i in range(2, n):
        next_fib = sequence[-1] + sequence[-2]
        sequence.append(next_fib)
        
    return sequence

if __name__ == '__main__':
    # Example usage: Get the first 10 Fibonacci numbers
    n_terms = 10
    try:
        result = fibonacci_sequence(n_terms)
        print(f"The first {n_terms} Fibonacci numbers are: {result}")
        
        # Example 2: Edge case
        print(f"Fibonacci for 1 term: {fibonacci_sequence(1)}")
        
        # Example 3: Edge case
        print(f"Fibonacci for 0 terms: {fibonacci_sequence(0)}")
        
    except ValueError as e:
        print(f"Error: {e}")
