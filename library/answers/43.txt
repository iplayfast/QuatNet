def to_dot(graph,digraph=True):
    e='->' if digraph else '--'
    s='digraph G {' if digraph else 'graph G {'
    s+='
'
    for u in graph.adj:
        for v,val in graph.adj[u]:
            label=f' [label="{val}"]' if val else ''
            s+=f'  "{u}" {e} "{v}"{label};
'
    s+='}'
    return s
