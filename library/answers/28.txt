    def __init__(self, name, age):
        self.name = name
        self.age = age

    def talk(self):
        return f"My name is {self.name}, I am {self.age} years old."

p = Person("Jane", 25)

import random

def generate_password(length=10, use_uppercase=True, use_numbers=True, use_special_chars=True):
    chars = "abcdefghijklmnopqrstuvwxyz"
    if use_uppercase:
        chars += "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
    if use_numbers:
        chars += "0123456789"
    if use_special_chars:
        chars += "!@#$%^&*()"

    password = "".join(random.choice(chars) for _ in range(length))
    return password

print(generate_password())
```

def binary_to_decimal(binary):
    decimal = 0
    power = 0
    for char in reversed(binary):
        if char == "1":
            decimal += 2 ** power
        power += 1
    return decimal

binary = "1010"
print(binary_to_decimal(binary))

def decimal_to_binary(decimal):
    binary = ""
    while decimal > 0:
        binary += str(decimal % 2)
        decimal //= 2
    return binary[::-1]

decimal = 10
print(decimal_to_binary(decimal))
```
