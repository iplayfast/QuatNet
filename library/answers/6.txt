class Node:
    def __init__(self, leaf=False):
        self.leaf = leaf
        self.keys = []
        self.children = []

def split(node):
    new_node = Node(leaf=node.leaf)
    mid = len(node.keys) // 2
    for key in node.keys[mid:]:
        new_node.keys.append(key)
    for child in node.children[mid:]:
        new_node.children.append(child)

    node.keys = node.keys[:mid]
    node.children = node.children[:mid]

    return new_node

def insert(root, key):
    if not root.leaf:
        i = 0
        while i < len(root.keys) and key > root.keys[i]:
            i += 1
        root.children.insert(i, insert(root.children[i], key))
    else:
        i = 0
        while i < len(root.keys) and key > root.keys[i]:
            i += 1
        if root.leaf and key == root.keys[i]:
            return False
        elif len(root.keys) == 2**(len(root.children)+1)-1:
            new_root = Node(leaf=False)
            new_root.children.append(Node(leaf=True))
            new_root.children.append(split(root))
            root = new_root

    if not root.leaf:
        return insert(root.children[0], key)

    if len(root.keys) < 2**(len(root.children)+1)-1:
        root.keys.insert(len(root.keys), key)
        return True
    else:
        return False

def search(root, key):
    i = 0
    while i < len(root.keys):
        if key == root.keys[i]:
            return True
        elif key > root.keys[i]:
            i += 1
        else:
            if not root.children[i].leaf:
                i = i * 2 + 1
            else:
                return False

    if not root.children[i].leaf:
        return search(root.children[i], key)

    return False

root = Node(leaf=True)
keys = [10, 20, 30, 40, 50]
for key in keys:
    insert(root, key)

print(search(root, 30))  # True
print(search(root, 90))  # False
