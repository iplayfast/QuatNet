```python
class RingBuffer:
    def __init__(self, size):
        self.size = size
        self.buffer = [None] * size
        self.head = 0
        self.tail = 0
        self.count = 0

    def produce(self, item):
        index = (self.head + self.count) % self.size
        self.buffer[index] = item
        self.count += 1

    def consume(self):
        if self.count == 0:
            return None
        index = self.tail
        item = self.buffer[index]
        self.buffer[index] = None
        self.tail = (self.tail + 1) % self.size
        self.count -= 1
        return item


# Example usage:
rb = RingBuffer(5)

rb.produce(1)
rb.produce(2)
rb.produce(3)
print(rb.consume())  # Output: 1
print(rb.consume())  # Output: 2
```
