def is_valid_integer(s: str) -> bool:
    """
    Checks if a string is a valid integer.
    Handles negative numbers, integers, and non-numeric input.
    """
    if not s:
        return False
    if not s.isdigit() or s == "-":
        return False
    for c in s:
        if not '0' <= c <= '9':
            return False
    return True
