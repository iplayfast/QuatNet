```python
import sys

def main():
    mode = input().strip()  # Read mode ('L' for linewise, 'C' for charwise)
    
    if mode.upper() == 'LIN':
        print('\n'.join(sys.stdin.read().splitlines()), end='')
    elif mode.upper() == 'CHAR':
        print(''.join(sys.stdin.read()), end='')

if __name__ == '__main__':
    main()
```
