def to_pig_latin(word):
    vowels = 'aeiouAEIOU'
    if word[0] in vowels:
        return word + 'yay'
    else:
        for i, letter in enumerate(word):
            if letter in vowels:
                return word[i:] + word[:i] + 'ay'
        return word + 'ay'  # In case the word has no vowels

def from_pig_latin(word):
    vowels = 'aeiouAEIOU'
    if word.endswith('yay'):
        return word[:-4] + word[-1] + 'ay'
    elif word.endswith('ay'):
        for i, letter in enumerate(word):
            if letter in vowels:
                return word[i:] + word[:i]
    return ''  # Return empty string if not a valid Pig Latin word

def translate(text):
    words = text.split()
    translated = [to_pig_latin(word) for word in words]
    return ' '.join(translated)

def reverse_translate(pig_text):
    words = pig_text.split()
    reversed = [from_pig_latin(word) for word in words]
    return ' '.join(reversed)
