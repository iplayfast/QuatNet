class DynamicList:
    def __init__(self):
        self.cap=16; self.data=[None]*16; self.size=0
    def add(self,x):
        if self.size>=self.cap:
            self._resize(self.cap*2)  # double capacity
        self.data[self.size]=x; self.size+=1
    def get(self,i):
        if i<0 or i>=self.size:
            raise IndexError
        return self.data[i]
    def remove(self,i):
        if i<0 or i>=self.size:
            raise IndexError
        for j in range(i,self.size-1):
            self.data[j]=self.data[j+1]
        self.size-=1
        if self.size>0 and self.size<=self.cap//4:
            self._resize(self.cap//2)  # halve when 25% full
    def _resize(self,new_cap):
        new=[None]*new_cap
        for i in range(self.size):
            new[i]=self.data[i]
        self.data=new; self.cap=new_cap
    def __len__(self): return self.size
# Doubling growth gives amortized O(1) append. Halving at 25% prevents thrashing.
