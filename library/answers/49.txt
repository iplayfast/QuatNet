def apply_func(lst, func):
    return [func(x) for x in lst]

def is_perfect_square(n):
    root = int(n**0.5)
    return root * root == n

perfect_squares = [i*i for i in range(1, 21)]
squared_perfect_squares = apply_func(perfect_squares, is_perfect_square)

for square in squared_perfect_squares:
    print(square)
