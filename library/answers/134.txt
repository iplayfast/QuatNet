```python
import math

def calculate_factorial(n: int) -> int:
    """
    Calculates the factorial of a non-negative integer n (n!).
    
    Args:
        n: The non-negative integer for which to calculate the factorial.
        
    Returns:
        The factorial of n.
    
    Raises:
        ValueError: If n is negative.
        TypeError: If n is not an integer.
    """
    if not isinstance(n, int):
        raise TypeError("Input must be an integer.")
    
    if n < 0:
        raise ValueError("Factorial is not defined for negative numbers.")
        
    if n == 0 or n == 1:
        return 1
    
    result = 1
    for i in range(2, n + 1):
        result *= i
    return result

if __name__ == "__main__":
    print("--- Factorial Calculator Test Cases ---")
    
    # Test case 1: Base case (0!)
    num1 = 0
    print(f"Factorial of {num1}: {calculate_factorial(num1)}")
    
    # Test case 2: Small positive number (5!)
    num2 = 5
    print(f"Factorial of {num2}: {calculate_factorial(num2)}")

    # Test case 3: Larger positive number (10!)
    num3 = 10
    print(f"Factorial of {num3}: {calculate_factorial(num3)}")

    # Test case 4: Error handling (Negative number)
    try:
        calculate_factorial(-3)
    except ValueError as e:
        print(f"\nError handling successful (Negative Input): {e}")
    
    # Test case 5: Error handling (Non-integer input)
    try:
        calculate_factorial(5.5)
    except TypeError as e:
        print(f"Error handling successful (Type Input): {e}")
```
