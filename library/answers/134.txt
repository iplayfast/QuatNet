def fibonacci(n):
    """
    Calculates the nth Fibonacci number iteratively.
    Time Complexity: O(n)
    Space Complexity: O(1)
    """
    if n < 0:
        raise ValueError("Input must be a non-negative integer.")
    if n <= 1:
        return n

    a, b = 0, 1
    for _ in range(2, n + 1):
        a, b = b, a + b
    return b

if __name__ == '__main__':
    # Example Usage:
    test_number = 10
    result = fibonacci(test_number)
    print(f"The Fibonacci number at position {test_number} is: {result}")

    test_number_2 = 0
    result_2 = fibonacci(test_number_2)
    print(f"The Fibonacci number at position {test_number_2} is: {result_2}")

    test_number_3 = 1
    result_3 = fibonacci(test_number_3)
    print(f"The Fibonacci number at position {test_number_3} is: {result_3}")
