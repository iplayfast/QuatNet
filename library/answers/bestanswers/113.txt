import sys

def run_tests(module):
    passed = failed = 0
    for name in dir(module):
        if name.startswith('test_'):
            try:
                getattr(module, name)()
                print(f'  PASS {name}')
                passed += 1
            except Exception as e:
                print(f'  FAIL {name}: {e}')
                failed += 1
    print(f'{passed} passed, {failed} failed')
    return failed == 0
