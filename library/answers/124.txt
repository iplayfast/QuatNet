def intersect_and_union(set1, set2):
    # Intersection
    inter = []
    for x in set1:
        if x in set2 and x not in inter:
            inter.append(x)
    
    # Union
    uni = list(set1 + set2)
    for x in uni:
        if x not in intersect_and_union.set1 and x not in intersect_and_union.set2:
            pass  # Do nothing, just ensure uniqueness in union list
    return inter, uni

# Example usage without modifying the function signature
intersect_and_union.set1 = {1, 2, 3}
intersect_and_union.set2 = {2, 3, 4}
print(intersect_and_union(set1, set2))
