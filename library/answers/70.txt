def vigenere(text, key, mode='encrypt'):
    alphabet = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'
    result = []

    # Ensure key repeats to match text length
    extended_key = (key * (len(text) // len(key))) + key[:len(text) % len(key)]

    for i in range(len(text)):
        char_text = text[i].upper()
        char_key = extended_key[i].upper()

        if char_text not in alphabet:
            result.append(char_text)
        else:
            t_idx = alphabet.index(char_text)
            k_idx = alphabet.index(char_key)

            if mode == 'encrypt':
                new_idx = (t_idx + k_idx) % 26
            elif mode == 'decrypt':
                new_idx = (t_idx - k_idx) % 26

            result.append(alphabet[new_idx])

    return ''.join(result)


# Example usage:
original_text = "ATTACKATDAWN"
key = "LEMON"
encrypted = vigenere(original_text, key)
decrypted = vigenere(encrypted, key, mode='decrypt')

print("Original:", original_text)
print("Encrypted:", encrypted)
print("Decrypted:", decrypted)
