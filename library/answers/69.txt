```python
import time

def exponential_backoff_retry(func, max_retries=3, base_delay=1):
    def wrapper(*args, **kwargs):
        retries = 0
        while retries < max_retries:
            try:
                return func(*args, **kwargs)
            except Exception as e:
                retry_after = base_delay * (2 ** retries)
                print(f"Attempt {retries + 1} failed with error: {e}. Retrying in {retry_after} seconds...")
                time.sleep(retry_after)
                retries += 1
        raise Exception(f"Max retries ({max_retries}) exceeded for function {func.__name__}")
    return wrapper

# Example usage:
@exponential_backoff_retry
def flaky_operation():
    import random
    if random.choice([True, False]):
        raise Exception("Flaky operation failed")
    return "Success"

print(flaky_operation())
```
