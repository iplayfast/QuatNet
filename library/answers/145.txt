import numpy as np

def dft(x):
    N = len(x)
    X = [0] * N
    for k in range(N):
        for n in range(N):
            angle = 2 * np.pi * k * n / N
            X[k] += x[n] * (np.cos(angle) - 1j * np.sin(angle))
    return X

# Example usage:
# signal = [0, 1, 0, -1]
# transformed = dft(signal)
