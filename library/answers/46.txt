def random_walk(graph,steps=20):
    import random
    nodes=list(graph.adj.keys())
    current=random.choice(nodes)
    result=[current]
    for _ in range(steps):
        neighbors=graph.adj.get(current,[])
        if neighbors:
            current=random.choice(neighbors)[0]
        else:
            current=random.choice(nodes)
        result.append(current)
    return ' '.join(result)
