def vigenere(text, key, mode='encrypt'):
    alphabet = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'
    result = []

    k_index = 0
    for char in text.upper():
        if char in alphabet:
            shift = alphabet.index(key[k_index % len(key)])
            if mode == 'encrypt':
                new_char = alphabet[(alphabet.index(char) + shift) % 26]
            elif mode == 'decrypt':
                new_char = alphabet[(alphabet.index(char) - shift) % 26]
            result.append(new_char)
            k_index += 1
        else:
            result.append(char)

    return ''.join(result)


# Example usage:
original_text = "ATTACKATDAWN"
key = "LEMON"

encrypted = vigenere(original_text, key)
decrypted = vigenere(encrypted, key, mode='decrypt')

print("Original:", original_text)
print("Encrypted:", encrypted)
print("Decrypted:", decrypted)
