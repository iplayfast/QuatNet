def winnowing(text, pattern, w=3, delta=2):
    def hash_function(substring):
        return sum(1 << (ord(c) - 32) for c in substring) % (10**6)

    pat_len = len(pattern)
    text_len = len(text)
    fingerprint_size = pat_len + w * (pat_len - 1)

    if fingerprint_size > text_len:
        return []

    fingerprints = {}
    for i in range(fingerprint_size):
        start = i
        end = start + pat_len
        substr = text[start:end]
        h = hash_function(substr)
        if h not in fingerprints:
            fingerprints[h] = [i, 0]
        else:
            fp_info = fingerprints[h]
            if end - start == pattern.count('*'):
                fp_info[1] += delta
            elif end - start < pattern.count('*'):
                del fingerprints[h]
        if fp_info[1] >= delta:
            yield substr

def detect_plagiarism(main_text, other_text, window=3, threshold=2):
    return set(winnowing(main_text, '***')) & set(winnowing(other_text, '***'))

# Example usage:
if __name__ == "__main__":
    main = "This is a sample text to detect plagiarism."
    other = "Another sample text with similar phrases for testing."
    common = detect_plagiarism(main, other)
    print("Common phrases:", common)
