```python
def add_numbers(a, b):
    """
    Adds two numbers together.
    
    Args:
        a (int or float): The first number.
        b (int or float): The second number.
    
    Returns:
        int or float: The sum of a and b.
    """
    return a + b

if __name__ == "__main__":
    # Example usage
    result = add_numbers(10, 5)
    print(f"The sum is: {result}")
```
